                            Joan Mendoza 5 months ago

                            Nice! and thanks! it worked perfectly https://plot.ly/~joan1094/0/

                            Advices in code:

                            for the server1.js:

                            var serialport = require("serialport");
                            var Serialport = serialport;
                            var portName = process.argv[2];

                            var myPort = new Serialport(portName, {
                            baudRate: 9600,
                            parser: serialport.parsers.readline("\r\n")
                            })

                            myPort.on('open',onOpen);
                            myPort.on('data',onData);

                            function onOpen(){
                            console.log("Open connection");
                            }

                            function onData(data){
                            //console.log("Temperatura tanque: "+data + " *C")
                            console.log(data)
                            }

                            You must call from the terminal: node server1.js "Arduino serialport". In my case was: node server1.js dev/ttyACM0
